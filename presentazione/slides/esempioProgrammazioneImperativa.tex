% Example of imperative programming in Picat
prova(X) =>
	if (X.length <= 3) then
		X[1] := -1
	else
		foreach (I in 1..X.length)
			X[I] := X[I] - 2
		end
	end.
main =>
	X := [],
	foreach(I in 1..5)
		X := X ++ [I]
	end,
	prova(X), println(X),
	X := [1,2,3], prova(X), println(X).
	\end{lstlisting}
